\begin{abstract}
We present a patched, publication-stable ``canonical spine'' for the MQGT--SCF program.
The framework extends GR+SM by two scalar singlets: a consciousness field $\Phic(x)$ and an
ethics field $\Eth(x)$. The two central hardening choices are: (i) measurement/collapse is modeled
as a local, covariant completely-positive trace-preserving (CPTP) dynamics in GKSL form
(Tomonaga--Schwinger hypersurface evolution), and (ii) ``teleology'' is not inserted as a
time-odd source term in the Lagrangian but instead implemented as a Radon--Nikod\'ym/Doob
tilt of the trajectory/outcome measure. We state explicit locality/microcausality conditions
yielding no-signaling, provide an energy-growth bound for collapse noise with finite bandwidth,
give a minimal EFT action with renormalizable couplings, and supply a conservative benchmark
parameter region for phenomenology. Remaining open items are isolated to operationalization of
the ethical functional and numerical constraint-fitting, rather than internal consistency.
\end{abstract}

\section{Patch notes: what was fixed and why}
MQGT--SCF drafts sometimes contain multiple co-equal implementations of ``teleology'' and collapse,
which makes the package feel internally inconsistent. This manuscript chooses one canonical route:

\begin{itemize}[leftmargin=*]
\item \textbf{Teleology:} canonical implementation is \emph{measure tilt} on paths/outcomes (Section~\ref{sec:teleology}),
not an explicit time-arrow term in $\Lag$.
\item \textbf{Measurement:} canonical implementation is \emph{local covariant GKSL} with microcausality
(Section~\ref{sec:measurement}), ensuring CPTP evolution and blocking superluminal signaling under stated assumptions.
\end{itemize}

Alternative legacy formulations are quarantined in Appendix~\ref{app:legacy}.

\section{Field content, symmetries, and EFT action}
\subsection{Degrees of freedom}
The minimal extension adds two real scalar singlets on spacetime $(M,g_{\mu\nu})$:
\[
\Phic(x)\quad\text{and}\quad \Eth(x),
\]
in addition to GR+SM fields.

\subsection{Canonical EFT action}
We work as an effective field theory (EFT) with local operators of canonical dimension $\le 4$.
The action is
\begin{equation}
S=\int \dd^4x\,\sqrt{-g}\;\Big(\Lag_{\rm GR}+\Lag_{\rm SM}+\Lag_{\Phic}+\Lag_{\Eth}+\Lag_{\rm int}\Big),
\label{eq:action}
\end{equation}
with
\begin{align}
\Lag_{\rm GR} &= \frac{\Mpl^2}{2}R-\Lambda,
\\
\Lag_{\Phic} &= -\frac12\,(\nabla \Phic)^2 - V_{\Phic}(\Phic), \qquad
V_{\Phic}(\Phic)=\frac12 m_{\Phic}^2 \Phic^2 + \frac{\lambda_{\Phic}}{4}\Phic^4,
\\
\Lag_{\Eth} &= -\frac12\,(\nabla \Eth)^2 - V_{\Eth}(\Eth), \qquad
V_{\Eth}(\Eth)=\frac12 m_{\Eth}^2 \Eth^2 + \frac{\lambda_{\Eth}}{4}\Eth^4.
\end{align}

The minimal renormalizable interactions are
\begin{equation}
\Lag_{\rm int}
=
-\frac{\kappa}{2}\Phic^2\Eth^2
-\gamma\,\Phic\,\Eth
-\alpha_H |H|^2 \Phic
-\beta_H |H|^2 \Eth
-g_{\Phic}\Phic\,\cO_m
-g_{\Eth}\Eth\,\cO_m.
\label{eq:int}
\end{equation}
Here $H$ is the SM Higgs doublet and $\cO_m$ is a coarse-grained matter/neural order parameter
(used only phenomenologically and restricted to one canonical observable family in this spine). The bilinear mixing term $\gamma\,\Phic\Eth$ is treated as a
standard interaction, not as a ``teleological arrow.''

\begin{remark}[Optional agent/inference term]
Some program documents include an auxiliary ``agentic'' or inference term $\Lag_{\rm Zora}$.
As long as it is not promoted to a nonlocal or non-Hermitian modification of the Hamiltonian density,
it can be treated as external bookkeeping and does not affect locality/gauge structure.
\end{remark}

\section{Classical microdynamics}
Varying \eqref{eq:action} gives the scalar equations of motion
\begin{align}
\square \Phic + m_{\Phic}^2 \Phic + \lambda_{\Phic}\Phic^3 + \kappa\,\Phic\,\Eth^2 + \gamma\,\Eth
&= J_{\Phic}(x),
\label{eq:eom_phic}
\\
\square \Eth + m_{\Eth}^2 \Eth + \lambda_{\Eth}\Eth^3 + \kappa\,\Eth\,\Phic^2 + \gamma\,\Phic
&= J_{\Eth}(x),
\label{eq:eom_eth}
\end{align}
where $J_{\Phic},J_{\Eth}$ encode effective sources from $\cO_m$ and SM portals.

\subsection{Canonical source-term choice (matter/brain coupling)}
To avoid multi-version drift, the canonical source model is fixed as
\begin{equation}
\cO_m(x)=0.7\,n(x)+0.3\,s(x),
\label{eq:Om}
\end{equation}
where $n(x)$ is a normalized neural-activity proxy and $s(x)$ a normalized somatic/systemic stress proxy,
both defined by the primary protocol family. Headline fits in this spine do not vary the $0.7/0.3$ split.
The source terms are
\begin{equation}
J_{\Phic}(x)=g_{\Phic}\,\cO_m(x)+\alpha_H |H(x)|^2,\qquad
J_{\Eth}(x)=g_{\Eth}\,\cO_m(x)+\beta_H |H(x)|^2.
\label{eq:canonical_sources}
\end{equation}
Equations \eqref{eq:Om}--\eqref{eq:canonical_sources} are the only main-text coupling choice used for
fits and simulations in this spine.

\subsection{Vacuum stability (boundedness)}
The quartic potential energy is
\[
V_4(\Phic,\Eth)=\frac{\lambda_{\Phic}}{4}\Phic^4+\frac{\lambda_{\Eth}}{4}\Eth^4+\frac{\kappa}{2}\Phic^2\Eth^2.
\]
A standard sufficient condition for boundedness is
\begin{equation}
\lambda_{\Phic}>0,\qquad \lambda_{\Eth}>0,\qquad \kappa>-\sqrt{\lambda_{\Phic}\lambda_{\Eth}}.
\label{eq:stability}
\end{equation}
The bilinear term $\gamma\Phic\Eth$ shifts the mass matrix and vacuum location but does not
change the large-field boundedness controlled by $V_4$.

\section{Quantization and measurement: covariant CPTP collapse}
\label{sec:measurement}

\subsection{Covariant GKSL in Tomonaga--Schwinger form}
Let $\Sigma$ be a Cauchy surface and $\rho[\Sigma]$ a density functional. The canonical
non-unitary dynamics is local GKSL on hypersurfaces:
\begin{equation}
\frac{\delta \rho[\Sigma]}{\delta \Sigma(x)}
=
-\frac{i}{\hbar}\Comm{\cH(x)}{\rho[\Sigma]}
+
\sum_a \left(
L_a(x)\rho[\Sigma]L_a^\dagger(x)
-\frac12\,\AComm{L_a^\dagger(x)L_a(x)}{\rho[\Sigma]}
\right).
\label{eq:TS_GKSL}
\end{equation}
The Lindblad densities may depend locally on $(\Phic(x),\Eth(x),\text{SM}(x))$.

\begin{definition}[Microcausality for collapse densities]
We impose
\begin{equation}
\Comm{L_a(x)}{L_b(y)}=0 \quad \text{whenever $x$ and $y$ are spacelike separated.}
\label{eq:microcausality}
\end{equation}
\end{definition}

\paragraph{No-signaling theorem package (v2 assumptions).}
We isolate the exact assumptions used by the theorem/proof chain:
\begin{itemize}[leftmargin=*]
\item \textbf{A1 (Local generator):} the Tomonaga--Schwinger generator density is a sum of local terms with support in bounded spacetime neighborhoods.
\item \textbf{A2 (Microcausality):} Eq.~\eqref{eq:microcausality} holds for all collapse densities.
\item \textbf{A3 (Linear CPTP dynamics):} the reduced evolution induced by \eqref{eq:TS_GKSL} defines a linear CPTP map on local algebras.
\item \textbf{A4 (Spacelike support separation):} Alice and Bob interventions are supported in spacelike-separated regions for the operational protocol window.
\end{itemize}

\begin{lemma}[Commuting local increments]
Under A1--A2, infinitesimal Tomonaga--Schwinger increments generated in spacelike-separated supports commute.
\end{lemma}

\textit{Proof sketch.}
The commutator of infinitesimal generators reduces to commutators of local Hamiltonian/Lindblad densities.
Microcausality annihilates these terms for spacelike-separated supports.

\begin{lemma}[Reduced-map closure]
Under A1--A3, taking a partial trace over Bob's algebra commutes with Alice-local evolution increments.
\end{lemma}

\textit{Proof sketch.}
Linearity and complete positivity ensure reduced maps are well-defined channels on Alice's subsystem; locality keeps Bob-supported increments outside Alice's reduced generator.

\begin{theorem}[Operational no-signaling under local covariant GKSL]
\label{thm:nosignaling_v2}
Under A1--A4, for any admissible Bob protocol in a spacelike-separated region, Alice's marginal outcome probabilities are invariant. Equivalently, the signaling capacity from Bob to Alice is zero over the protocol window.
\end{theorem}

\textit{Proof sketch.}
By Lemma 1, spacelike-separated increments are order-independent. By Lemma 2, Bob-local increments vanish after reduction to Alice's algebra. Therefore the Alice marginal channel is independent of Bob's intervention choice, yielding protocol-level no-signaling.

\begin{remark}[Scope boundary]
The theorem is operational and conditional on A1--A4. Violating locality (nonlocal Lindblad densities), microcausality, or linear CPTP structure invalidates the conclusion. This boundary is explicit and test-auditable.
\end{remark}

\subsection{Equivalent foliation form and a concrete local collapse operator}
In a chosen foliation (useful for calculations), a standard local dissipator is the double-commutator form
\begin{equation}
\dot{\rho}
=
-\frac{i}{\hbar}\Comm{H}{\rho}
-\frac{\gamma_{\rm col}}{2}
\int \dd^3x\;\Comm{L(x)}{\Comm{L(x)}{\rho}},
\qquad
L(x)=\sqrt{f(\Phic(x),\Eth(x))}\;\hat{M}(x),
\label{eq:double_comm}
\end{equation}
where $\hat{M}(x)$ is a local Hermitian density (e.g.\ mass density) and $f(\Phic,\Eth)\ge 0$
is a classical scalar functional. This is CPTP for $\gamma_{\rm col}\ge 0$.

\paragraph{Two textbook solutions.}
(i) Two-level dephasing: with $\hat{M}=\sigma_z$ and constant $f$ over the support,
coherences decay exponentially while populations remain fixed. \\
(ii) Free particle localization: in the position basis one obtains Gaussian suppression of off-diagonals,
$\rho(x,x',t)\propto \exp(-\Gamma t\,|x-x'|^2)$ with $\Gamma\propto \gamma_{\rm col}\,\overline{f}$.

\subsection{Bounded heating under colored/dissipative noise}
Collapse noise can inject energy; a minimal hardening requirement is that the energy growth rate
is finite and tunably small.

\begin{proposition}[Bounded energy growth]
Assume the effective collapse noise has finite correlation time, equivalently a spectral density
$S(\omega)$ of finite area. Then the average energy growth rate under \eqref{eq:double_comm} is finite and can be made
$<\varepsilon$ by band-limiting $S(\omega)$ and/or adding a dissipative Lindblad term that thermalizes high-frequency modes.
\end{proposition}

\textit{Proof sketch.}
Compute $\frac{\dd}{\dd t}\langle H\rangle=\Tr(H\dot{\rho})$ and expand in an energy eigenbasis. The rate reduces to an
integral over $\omega\,S(\omega)$ weighted by transition matrix elements. Finite spectral area implies convergence,
and the dissipator bounds high-$\omega$ contributions.

\section{Teleology in the measure (no source in the Lagrangian)}
\label{sec:teleology}

\subsection{Two-layer construction}
The canonical ``teleology'' implementation is:
\begin{enumerate}[leftmargin=*,label=(\arabic*)]
\item \textbf{CPT-symmetric microdynamics:} the generator is unitary drift plus CPTP local dissipators
(e.g.\ \eqref{eq:TS_GKSL} or \eqref{eq:double_comm}).
\item \textbf{Measure tilt:} preference enters by reweighting the probability measure on trajectories/outcomes.
\end{enumerate}

\subsection{Radon--Nikod\'ym/Doob tilt on path space}
Let $\Gamma$ denote a system trajectory under a baseline law $P_0$ (e.g.\ an unraveling of the GKSL dynamics).
Define a local, additive ethical functional
\begin{equation}
J_E[\Gamma]=\int_{0}^{T} j_E(\Gamma_t)\,\dd t,
\label{eq:JE}
\end{equation}
where $j_E$ is a bounded local functional of the trajectory state and local fields (including $\Phic,\Eth$).
For $\eta\in\mathbb{R}$, define the tilted law
\begin{equation}
\frac{\dd P_\eta}{\dd P_0}[\Gamma]
=
\exp\Big(\eta J_E[\Gamma]-\psi(\eta)\,T\Big),
\label{eq:RN_tilt}
\end{equation}
with $\psi(\eta)$ chosen to normalize $P_\eta$.

\subsection{Ethics-weighted Born rule (local form)}
Consider a POVM $\{M_i\}$ in a spacetime region $R$ with baseline weights
$p_i=\langle \Psi|M_i^\dagger M_i|\Psi\rangle$ (or $p_i=\Tr(\rho\,M_i^\dagger M_i)$). Let $E_i$ denote the coarse-grained
ethical score conditional on outcome $i$ in $R$ (an estimator derived from $J_E$ restricted to $R$).
Then the tilted selection rule is
\begin{equation}
P_\eta(i\mid R)=
\frac{p_i\,e^{\eta E_i}}{\sum_j p_j\,e^{\eta E_j}}.
\label{eq:weighted_born}
\end{equation}
For $|\eta|\ll 1$, $P_\eta(i)=p_i+\mathcal{O}(\eta)$ with explicit first-order correction by $(E_i-\langle E\rangle)$.

\begin{remark}[Canonical deformation-law convention]
\label{rem:canonical_law}
Equations \eqref{eq:weighted_born} and \eqref{eq:Ei_from_delta} define the only headline probability law used in this spine:
\[
P_\eta(i\mid R)=\frac{p_i\,e^{\eta E_i}}{\sum_j p_j\,e^{\eta E_j}},
\qquad
E_i=\frac{\Delta E_i}{C_E}.
\]
The sign convention is fixed so that positive $\eta$ favors larger $E_i$.
Standard quantum mechanics is recovered in any of the limits $\eta\to 0$, $\Delta E_i\to 0$, or $C_E\to\infty$.
Legacy linearized or sign-flipped forms are approximation conventions only and are not canonical.
\end{remark}

\subsection{Operational ethics estimator (lab-computable)}
To remove interpretive freedom, we define the ethical estimator in measurable steps:
\begin{enumerate}[leftmargin=*,label=(\alph*)]
\item Choose observables only from the canonical primary family
\[
O_1(x,t)=n(x,t), \qquad O_2(x,t)=s(x,t),
\]
with fixed units and fixed weights $(w_1,w_2)=(0.7,0.3)$.
\item Define local ethical density
\begin{equation}
e(x,t)=0.7\,n(x,t)+0.3\,s(x,t),
\label{eq:local_ethics_density}
\end{equation}
with units reported explicitly in the protocol sheet.
\item Define aggregate ethical resonance over analysis region $V$:
\begin{equation}
E_R(t)=\int_V e(x,t)\,\dd^3x.
\label{eq:ER}
\end{equation}
\item For outcome branch $a$, define
\begin{equation}
\Delta E_a = E_R^{(a)}(t_*)-E_R^{\rm baseline}(t_*),
\label{eq:deltaEa}
\end{equation}
where $t_*$ is the preregistered scoring horizon.
\item Map branch scores to weighted Born factors by setting
\begin{equation}
E_i \equiv \frac{\Delta E_i}{C_E},
\label{eq:Ei_from_delta}
\end{equation}
with fixed $C_E$ from calibration runs.
\end{enumerate}

\begin{remark}[Operational boundary]
Equations \eqref{eq:local_ethics_density}--\eqref{eq:Ei_from_delta} define the canonical estimator for this spine.
Alternative estimators are allowed only as appendix-only sensitivity analyses and must not replace the primary preregistered map.
\end{remark}

\section{One hard quantitative handle: interferometric coherence loss}
From the Gaussian suppression behavior implied by \eqref{eq:double_comm}, a minimal falsifier is an
additional decoherence of spatial superpositions:
\begin{equation}
\frac{V}{V_0}\;\approx\;\exp\!\Big(-\Gamma\,T\,\Delta x^2\Big),\qquad
\Gamma \sim \gamma_{\rm col}\,\overline{f(\Phic,\Eth)}\times(\text{apparatus factor}),
\label{eq:visibility}
\end{equation}
where $V$ is observed visibility, $T$ the superposition time, and $\Delta x$ the path separation.
A null result at sensitivity $\delta$ yields a bound
\[
\Gamma \lesssim \frac{|\ln(1-\delta)|}{T\,\Delta x^2},
\]
which directly constrains $\gamma_{\rm col}\,\overline{f(\Phic,\Eth)}$.

\appendix
\section{Legacy teleology-in-$\Lag$ term (deprecated option)}
\label{app:legacy}
Some drafts include an explicit bilinear ``teleology'' term of the form
$\Lag_{\rm teleo}=-\xi\,\Phic\,\Eth$ and interpret it as a future-directed drive.
In a conservative Hamiltonian field theory, such a potential term is time-reversal even for scalar
fields and does not itself generate an arrow of time without dissipation or boundary conditions.
In the patched canonical spine, any bilinear term is absorbed into the ordinary interaction
$\gamma\,\Phic\,\Eth$, and ``teleology'' is reserved for the measure-tilt layer \eqref{eq:RN_tilt}.
